    BrinsonFachlerService.compute(trades: list[ClosedTrade]) -> list[AttributionLevel]:
      Level 1 "portfolio": total portfolio return vs 0 (benchmark unavailable without DataClient; use 0 as placeholder)
      Level 2 "strategy": split trades by strategy ("swing"/"position"), compute per-strategy avg pnl_pct, allocation = relative weight, selection = excess return
      Level 3 "trade": per-trade excess return vs strategy avg; allocation_effect = 0 (same level), selection_effect = trade.pnl_pct - strategy_avg
      Level 4 "skill": use IC to proxy skill; allocation_effect = composite_score contribution, selection_effect = IC * avg_pnl_pct
      Each level: total_effect = allocation_effect + selection_effect + interaction_effect

    ICCalculationService:
      MIN_TRADES_FOR_IC = 20
      compute_axis_ic(trades, axis) -> Optional[float]:
        if len(trades) < MIN_TRADES_FOR_IC: return None
        scores = [getattr(t, f"{axis}_score") for t in trades if getattr(t, f"{axis}_score") is not None]
        returns = [t.pnl_pct for t in trades if getattr(t, f"{axis}_score") is not None]
        if len(scores) < MIN_TRADES_FOR_IC or all(s == scores[0] for s in scores): return None
        from scipy.stats import spearmanr
        ic, _ = spearmanr(scores, returns)
        return round(float(ic), 4) if not math.isnan(ic) else None
