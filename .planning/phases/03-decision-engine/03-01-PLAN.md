         SELECT s.symbol, s.composite_score, s.risk_adjusted_score, v.intrinsic_value, v.margin_of_safety, v.has_margin, sig.direction, sig.strength
         FROM scores s LEFT JOIN valuations v ON s.symbol = v.symbol LEFT JOIN signals sig ON s.symbol = sig.symbol
         WHERE s.risk_adjusted_score >= min_composite, optional AND sig.direction = signal_filter
         ORDER BY s.risk_adjusted_score DESC LIMIT top_n.
