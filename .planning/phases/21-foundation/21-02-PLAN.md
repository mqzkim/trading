**Step 1: Configure next.config.ts with API rewrites.**

Replace the generated `next.config.ts` with:
```typescript
import type { NextConfig } from 'next';

const FASTAPI_URL = process.env.FASTAPI_URL || 'http://127.0.0.1:8000';

const nextConfig: NextConfig = {
  async rewrites() {
    return [
      {
        source: '/api/:path*',
        destination: `${FASTAPI_URL}/api/:path*`,
      },
    ];
  },
};

export default nextConfig;
```

The single wildcard rule `/api/:path*` matches all API paths including `/api/v1/dashboard/events` (SSE). No need for a separate SSE rewrite rule -- the wildcard covers it. `FASTAPI_URL` env var allows override but defaults to `http://127.0.0.1:8000`.

**Step 4: Create 4 page stubs as Client Components.**

Each page is a `'use client'` component that:
1. Fetches data from its JSON API endpoint via `fetch()` in `useEffect`
2. Displays raw JSON in a `<pre>` block (real UI is Phase 22)
3. Shows an error message if FastAPI is unreachable: "Backend connection failed - Start FastAPI first"
4. Shows a loading state while fetching

Page files and their fetch URLs:
- `src/app/(dashboard)/page.tsx` -> `fetch('/api/v1/dashboard/overview')`
- `src/app/(dashboard)/signals/page.tsx` -> `fetch('/api/v1/dashboard/signals')`
- `src/app/(dashboard)/risk/page.tsx` -> `fetch('/api/v1/dashboard/risk')`
- `src/app/(dashboard)/pipeline/page.tsx` -> `fetch('/api/v1/dashboard/pipeline')`

Pattern for each page (adapt title and URL):
```typescript
'use client';
import { useEffect, useState } from 'react';

export default function OverviewPage() {
  const [data, setData] = useState<Record<string, unknown> | null>(null);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    fetch('/api/v1/dashboard/overview')
      .then((res) => {
        if (!res.ok) throw new Error(`Backend error: ${res.status}`);
        return res.json();
      })
      .then(setData)
      .catch((err) => setError(err.message));
  }, []);

  if (error) {
    return (
      <div className="p-8 text-red-400">
        Backend connection failed - Start FastAPI first: {error}
      </div>
    );
  }
  if (!data) return <div className="p-8 text-gray-400">Loading...</div>;

  return (
    <div>
      <h1 className="text-xl font-bold mb-4">Overview</h1>
      <pre className="bg-gray-900 p-4 rounded text-sm overflow-auto text-gray-300">
        {JSON.stringify(data, null, 2)}
      </pre>
    </div>
  );
}
```
