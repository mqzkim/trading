```python
def get_vix() -> float:
    df = yf.Ticker("^VIX").history(period="5d")
    return float(df["Close"].iloc[-1])

def get_sp500_vs_200ma() -> float:
    df = yf.Ticker("^GSPC").history(period="2y")
    close = df["Close"]; ma200 = close.rolling(200).mean()
    return float(close.iloc[-1] / ma200.iloc[-1])

def get_yield_curve_slope() -> float:
    ten_year = yf.Ticker("^TNX").history(period="5d")["Close"].iloc[-1]
    three_month = yf.Ticker("^IRX").history(period="5d")["Close"].iloc[-1]
    return float((ten_year - three_month) * 100)
```
